# REQUIRE LEAST CMAKE 3.10.3
CMAKE_MINIMUM_REQUIRED(VERSION 3.10.3)

LIST(APPEND CMAKE_MODULE_PATH ${CMAKE_HOME_DIRECTORY}/cmake/Concrete/modules)

INCLUDE(Concrete)

PROJECT(${CONCRETE_PROJECT_DEFAULT_PARAMETER})

CONCRETE_METHOD_PROJECT_CONFIGURE(
    PROJECT_NAME         libhdc
    PROJECT_VERSION      1 1 0
    PROJECT_DESCRIPTION  "hardware information collect library"
    PROJECT_HOMEPAGE_URL "https://github.com/FrankPIE/SMBIOS"
    PROJECT_LANGUAGES    C ASM_MASM

    PROJECT_CONFIGURATION_TYPES Debug Release

    PROJECT_ROOT_DIR            ${CMAKE_HOME_DIRECTORY}/
    PROJECT_BINARY_OUTPUT_DIR   ${CMAKE_HOME_DIRECTORY}/bin
    PROJECT_LIBRARY_OUTPUT_DIR  ${CMAKE_HOME_DIRECTORY}/lib
    PROJECT_COMPILER_TARGET_SUBDIRECTORY
)

CONCRETE_METHOD_SET_GLOBAL_PROPERTIES(
    PROPERTIES 
        USE_FOLDERS  ON
        PREDEFINED_TARGETS_FOLDER "ConcreteTargets"
)

CONCRETE_METHOD_GLOBAL_TARGET_CONFIGURE(
    INCLUDE_DIRECTORIES 
        ${CMAKE_BINARY_DIR}
)

CONCRETE_METHOD_ADD_SUBDIRECTORYS(
    SOURCE_DIRECTORYS 
        ${CMAKE_HOME_DIRECTORY}/src 
        ${CMAKE_HOME_DIRECTORY}/example
    BINARY_DIRECTORIES
        libhdc.project
        example.project
    FOLDERS
        "Library"
        "Example"
)

CONCRETE_METHOD_SET_VS_STARTUP_PROJECT(HashMap)
